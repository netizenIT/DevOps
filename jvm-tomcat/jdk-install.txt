Install by script
---------------------------------------------------------------------------------------------------------------------
#!/bin/bash
yum update -y;
yum install -y wget;
mkdir /usr/share/jvm;
cd /usr/share/jvm;
wget --no-cookies --no-check-certificate --header "Cookie: oraclelicense=accept-securebackup-cookie" \
http://download.oracle.com/otn-pub/java/jdk/8u191-b12/2787e4a523244c269598db4e85c51e0c/jdk-8u191-linux-x64.tar.gz \
-O jdk-8_linux-x64_bin.tar.gz;
tar -xzf jdk-8_linux-x64_bin.tar.gz;echo -e "#!/bin/bash
JAVA_HOME=/usr/share/jvm/jdk1.8.0_191
export JAVA_HOME
PATH=\$JAVA_HOME/bin:\$PATH
export PATH
" > /etc/profile.d/jdk_set_home.sh
